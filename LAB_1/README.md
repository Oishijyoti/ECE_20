    return `${first_name} ${last_name}`;
}

const first_name = "John";
const last_name = "Doe";
console.log(get_full_name(first_name, last_name));
// in camel case

const firstName = "John";
const lastName = "Doe";

function getFullName(firstName, lastName) {
    return `${firstName} ${lastName}`;
}
